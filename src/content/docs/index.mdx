## Examples

See at a glance what Pomsky looks like.

<Tabs>
  <TabItem label="Pomsky">

    ```pomsky
    'Hello' [space]+ ('world' | 'pomsky')
    ```

  </TabItem>
  <TabItem label="Regex">

    ```regexp
    Hello\s+(?:world|pomsky)
    ```

  </TabItem>
</Tabs>

### Boundaries

<Tabs>
  <TabItem label="Pomsky">

    ```pomsky
    ^ % 'foo' % $
    ```

  </TabItem>
  <TabItem label="Regex">

    ```regexp
    ^\bfoo\b$
    ```

  </TabItem>
</Tabs>

### Repetition

<Tabs>
  <TabItem label="Pomsky">

    ```pomsky
    'hello'{1,5} 'world'*
    ```

  </TabItem>
  <TabItem label="Regex">

    ```regexp
    (?:hello){1,5}(?:world)*
    ```

  </TabItem>
</Tabs>

### Character classes

<Tabs>
  <TabItem label="Pomsky">

    ```pomsky
    ['acd[]' 'p'-'s' digit]
    !['^']  # negated
    ```

  </TabItem>
  <TabItem label="Regex">

    ```regexp
    [acd[\]p-s\d][^\^]
    ```

  </TabItem>
</Tabs>

### Unicode

<Tabs>
  <TabItem label="Pomsky">

    ```pomsky
    [s U+15 U+FEFF Greek Decimal_Number]
    ```

  </TabItem>
  <TabItem label="Regex">

    ```regexp
    [\s\x15\uFEFF\p{Greek}\p{Nd}]
    ```

  </TabItem>
</Tabs>

### Capturing groups

<Tabs>
  <TabItem label="Pomsky">

    ```pomsky
    :([space]) :named(.)
    ```

  </TabItem>
  <TabItem label="Regex">

    ```regexp
    (\s)(?<named>.)
    ```

  </TabItem>
</Tabs>

### Lookaround

<Tabs>
  <TabItem label="Pomsky">

    ```pomsky
    (<< 'foo' | 'bar')
    (!>> 'foo' | 'bar')
    ```

  </TabItem>
  <TabItem label="Regex">

    ```regexp
    (?<=foo|bar)(?!foo|bar)
    ```

  </TabItem>
</Tabs>

### Ranges

<Tabs>
  <TabItem label="Pomsky">

    ```pomsky
    range '0'-'255'
    ```

  </TabItem>
  <TabItem label="Regex">

    ```regexp
    0|1[0-9]{0,2}|2(?:[0-4][0-9]?|5[0-5]?|[6-9])?|[3-9][0-9]?
    ```

  </TabItem>
</Tabs>

### Variables

<Tabs>
  <TabItem label="Pomsky">

    ```pomsky
    let number = '-'? [digit]+;
    let op = ['+-*/'];
    number (op number)*
    ```

  </TabItem>
  <TabItem label="Regex">

    ```regexp
    -?\d+(?:[+\-*/]-?\d+)*
    ```

  </TabItem>
</Tabs>
